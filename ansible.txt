vi /etc/netplan/00-installer-config.yaml
-----
network:
  ethernets:
    eth0:
      addresses:
      - 192.168.1.x/24
      gateway4: 192.168.1.1
      nameservers:
        addresses:
        - 192.168.1.1
        search:
        - uz.local
  version: 2
-----
reboot

/dev/sdb1       /gfs/infra   ext4    defaults,nofail
